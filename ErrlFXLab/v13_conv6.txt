5. Sidebar nav behaviors

Under your VS helpers or near the bottom with other button listeners, add:

    // ---- Sidebar "Views" navigation ----
    function smoothScrollTo(element) {
      if (!element) return;
      element.scrollIntoView({
        behavior: "smooth",
        block: "nearest",
      });
    }

    function flashSectionBorder(el) {
      if (!el) return;
      el.classList.add("ring-2", "ring-cyan-400/80");
      setTimeout(() => {
        el.classList.remove("ring-2", "ring-cyan-400/80");
      }, 500);
    }

    navGalleryBtn.addEventListener("click", () => {
      smoothScrollTo(capturedFramesSection);
      flashSectionBorder(capturedFramesSection);
      hintLabel.textContent = "Gallery: click frames to download ¬∑ Shift = VS Left ¬∑ Alt = VS Right.";
      playBeep(620, 0.07, "triangle");
    });

    navFavoritesBtn.addEventListener("click", () => {
      smoothScrollTo(capturedFramesSection);
      flashSectionBorder(capturedFramesSection);
      // Briefly highlight favorite frames
      const favorites = capturedFrames.filter((f) => f.favorite && f.el);
      if (!favorites.length) {
        vsStatusEl.textContent = "No favorites yet. Click ‚òÖ on frames to mark them.";
        flashStatus("Mark frames as ‚òÖ favorites to use this view.", "info");
        playBeep(300, 0.08, "square");
        return;
      }

      favorites.forEach((f) => {
        f.el.classList.add("ring-2", "ring-amber-400/80");
        setTimeout(() => {
          f.el.classList.remove("ring-2", "ring-amber-400/80");
        }, 900);
      });

      vsStatusEl.textContent = "Favorites highlighted in gallery.";
      hintLabel.textContent = "Use 'Use first 2 favorites' below VS to auto-fill both slots.";
      playBeep(780, 0.07, "triangle");
    });

    navVsBtn.addEventListener("click", () => {
      smoothScrollTo(vsSection);
      flashSectionBorder(vsSection);
      hintLabel.textContent = "VS: Shift-click = Left, Alt-click = Right, then 'Download VS image'.";
      playBeep(700, 0.07, "triangle");
    });

Now the left sidebar is actually a navigation panel:
	‚Ä¢	Gallery ‚Üí scroll to thumbnails, flash border, update hint
	‚Ä¢	Favorites ‚Üí scroll to gallery, briefly highlight ‚òÖ thumbs
	‚Ä¢	VS ‚Üí scroll to VS block, flash border, hint about usage

6. Keyboard shortcuts for tab switching

We already have a big window.addEventListener("keydown"... that handles:
	‚Ä¢	Space ‚Üí pause/play p5
	‚Ä¢	1 / 2 ‚Üí cycle presets
	‚Ä¢	c ‚Üí capture frame
	‚Ä¢	s ‚Üí save FX preset

Let‚Äôs extend that to also switch preview tabs:
	‚Ä¢	q = SVG tab
	‚Ä¢	w = p5 tab
	‚Ä¢	e = HTML FX tab
	‚Ä¢	r = 3D Wire tab
	‚Ä¢	t = Voxels tab
	‚Ä¢	g = Gallery section
	‚Ä¢	v = VS section

First, we want a tiny helper to programmatically ‚Äúclick‚Äù a tab button. Above the keydown handler, add:

    function activatePreviewTab(tabName) {
      const btn = Array.from(tabButtons).find((b) => b.dataset.tab === tabName);
      if (btn) {
        btn.click();
      }
    }

Then, in your window.addEventListener("keydown", (event) => { ... }), extend the key handling.

You already have:

      const key = event.key.toLowerCase();

      // Space: pause/play
      if (event.code === "Space") { ... }

      // 1/2: cycle p5 presets
      if (key === "1" || key === "2") { ... }

      // C: capture (p5)
      if (key === "c") { ... }

      // S: quick-save FX preset pack
      if (key === "s") { ... }

Right after the key line, insert the tab shortcuts:

      const key = event.key.toLowerCase();

      // Tab shortcuts: Q/W/E/R/T switch preview tabs
      if (key === "q") {
        activatePreviewTab("svg");
        return;
      }
      if (key === "w") {
        activatePreviewTab("p5");
        return;
      }
      if (key === "e") {
        activatePreviewTab("htmlfx");
        return;
      }
      if (key === "r") {
        activatePreviewTab("wire3d");
        return;
      }
      if (key === "t") {
        activatePreviewTab("voxel");
        return;
      }

      // Section shortcuts: G = Gallery, V = VS section
      if (key === "g") {
        smoothScrollTo(capturedFramesSection);
        flashSectionBorder(capturedFramesSection);
        hintLabel.textContent = "Gallery via keyboard ¬∑ Shift thumb = Left, Alt = Right.";
        playBeep(640, 0.07, "triangle");
        return;
      }
      if (key === "v") {
        smoothScrollTo(vsSection);
        flashSectionBorder(vsSection);
        hintLabel.textContent = "VS via keyboard ¬∑ fill slots, then export.";
        playBeep(720, 0.07, "triangle");
        return;
      }

The rest of the key handling (space / 1 / 2 / c / s) stays exactly as-is, just below those blocks.
