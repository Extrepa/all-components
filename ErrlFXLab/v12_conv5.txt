4. VS slot helpers (assign + render)

Below the thumbnail helpers (after captureCurrentFrameToGallery()), add:

    // ---- VS Mode helpers ----
    function renderVsSlots() {
      // Left
      if (vsLeftFrame && vsLeftImg) {
        vsLeftImg.src = vsLeftFrame.url;
        vsLeftImg.classList.remove("hidden");
        // Hide placeholder text
        const placeholders = vsLeftSlot.querySelectorAll("span.text-[10px].text-slate-500");
        placeholders.forEach((el) => el.classList.add("hidden"));
      } else if (vsLeftSlot && vsLeftImg) {
        vsLeftImg.src = "";
        vsLeftImg.classList.add("hidden");
        const placeholders = vsLeftSlot.querySelectorAll("span.text-[10px].text-slate-500");
        placeholders.forEach((el) => el.classList.remove("hidden"));
      }

      // Right
      if (vsRightFrame && vsRightImg) {
        vsRightImg.src = vsRightFrame.url;
        vsRightImg.classList.remove("hidden");
        const placeholders = vsRightSlot.querySelectorAll("span.text-[10px].text-slate-500");
        placeholders.forEach((el) => el.classList.add("hidden"));
      } else if (vsRightSlot && vsRightImg) {
        vsRightImg.src = "";
        vsRightImg.classList.add("hidden");
        const placeholders = vsRightSlot.querySelectorAll("span.text-[10px].text-slate-500");
        placeholders.forEach((el) => el.classList.remove("hidden"));
      }
    }

    function assignVsSlot(frame, side) {
      if (!frame) return;
      if (side === "left") {
        vsLeftFrame = frame;
        renderVsSlots();
        vsStatusEl.textContent = "Left slot set from thumbnail.";
        flashStatus("VS Left set ‚úÖ");
        playBeep(640, 0.08, "triangle");
      } else if (side === "right") {
        vsRightFrame = frame;
        renderVsSlots();
        vsStatusEl.textContent = "Right slot set from thumbnail.";
        flashStatus("VS Right set ‚úÖ");
        playBeep(760, 0.08, "triangle");
      }
    }

    function clearVsSlots() {
      vsLeftFrame = null;
      vsRightFrame = null;
      renderVsSlots();
      vsStatusEl.textContent = "VS slots cleared.";
      playBeep(260, 0.08, "square");
    }

6. Download combined VS image

This is the ‚Äúshareable artifact‚Äù part.
We‚Äôll composite the left and right frames into one side-by-side PNG and download it.

Add:

    vsDownloadBtn.addEventListener("click", () => {
      if (!vsLeftFrame || !vsRightFrame) {
        vsStatusEl.textContent = "Set both Left and Right before downloading.";
        flashStatus("VS needs both sides filled.", "error");
        playBeep(200, 0.1, "square");
        return;
      }

      const canvas = document.createElement("canvas");
      const width = 900;
      const height = 450;
      canvas.width = width;
      canvas.height = height;
      const ctx = canvas.getContext("2d");
      if (!ctx) {
        vsStatusEl.textContent = "Canvas not supported for VS export.";
        return;
      }

      ctx.fillStyle = "#02070a";
      ctx.fillRect(0, 0, width, height);

      const leftImg = new Image();
      const rightImg = new Image();

      let loaded = 0;
      function onLoaded() {
        loaded++;
        if (loaded < 2) return;

        // Draw "cards"
        const padding = 20;
        const cardWidth = (width - padding * 3) / 2;
        const cardHeight = height - padding * 2;

        function drawCard(img, x) {
          // Frame
          ctx.save();
          const radius = 24;
          const y = padding;

          // Card background with gradient
          const grad = ctx.createLinearGradient(x, y, x + cardWidth, y + cardHeight);
          grad.addColorStop(0, "#071b25");
          grad.addColorStop(1, "#02070a");

          // Rounded rect
          ctx.beginPath();
          ctx.moveTo(x + radius, y);
          ctx.lineTo(x + cardWidth - radius, y);
          ctx.quadraticCurveTo(x + cardWidth, y, x + cardWidth, y + radius);
          ctx.lineTo(x + cardWidth, y + cardHeight - radius);
          ctx.quadraticCurveTo(
            x + cardWidth,
            y + cardHeight,
            x + cardWidth - radius,
            y + cardHeight
          );
          ctx.lineTo(x + radius, y + cardHeight);
          ctx.quadraticCurveTo(x, y + cardHeight, x, y + cardHeight - radius);
          ctx.lineTo(x, y + radius);
          ctx.quadraticCurveTo(x, y, x + radius, y);
          ctx.closePath();

          ctx.fillStyle = grad;
          ctx.fill();

          ctx.strokeStyle = "#34e1ff";
          ctx.lineWidth = 2;
          ctx.stroke();

          // Inner image, contained
          const innerPadding = 16;
          const innerX = x + innerPadding;
          const innerY = y + innerPadding;
          const innerW = cardWidth - innerPadding * 2;
          const innerH = cardHeight - innerPadding * 2;

          const imgRatio = img.width / img.height;
          const boxRatio = innerW / innerH;

          let drawW, drawH;
          if (imgRatio > boxRatio) {
            drawW = innerW;
            drawH = innerW / imgRatio;
          } else {
            drawH = innerH;
            drawW = innerH * imgRatio;
          }

          const dx = innerX + (innerW - drawW) / 2;
          const dy = innerY + (innerH - drawH) / 2;

          ctx.drawImage(img, dx, dy, drawW, drawH);
          ctx.restore();
        }

        // Left and right cards
        drawCard(leftImg, padding);
        drawCard(rightImg, padding * 2 + cardWidth);

        // Center VS label
        ctx.fillStyle = "#ecf9ff";
        ctx.font = "bold 32px system-ui, -apple-system, BlinkMacSystemFont, sans-serif";
        ctx.textAlign = "center";
        ctx.textBaseline = "middle";
        ctx.fillText("VS", width / 2, height / 2);

        ctx.font = "12px system-ui, -apple-system, BlinkMacSystemFont, sans-serif";
        ctx.fillStyle = "#8eb7c7";
        ctx.fillText("Errl FX VS export", width / 2, height - 18);

        const dataUrl = canvas.toDataURL("image/png");
        const a = document.createElement("a");
        a.href = dataUrl;
        a.download = "errl-fx-vs.png";
        a.click();

        vsStatusEl.textContent = "VS image exported.";
        flashStatus("VS image downloaded ‚öîÔ∏è");
        playBeep(840, 0.1, "triangle");
      }

      leftImg.onload = onLoaded;
      rightImg.onload = onLoaded;
      leftImg.onerror = () => {
        vsStatusEl.textContent = "Failed to load left frame for VS export.";
      };
      rightImg.onerror = () => {
        vsStatusEl.textContent = "Failed to load right frame for VS export.";
      };

      leftImg.src = vsLeftFrame.url;
      rightImg.src = vsRightFrame.url;
    });
